31       private void clearTextFields(GridPane grid) {
32           // Simple method to clear the text fields in a grid
33           for (Node node : grid.getChildren()) {
34               if (node instanceof TextField) {
35                   // clear
36                   ((TextField)node).setText("");
37                   node.setStyle(null);
38               }
39           }
40       }

42       private void resetTextFields(GridPane grid) {
43           // Simple method to reset the style on text fields in a grid which have a style set
44           // and don't touch the other fields
45           // We use it to clear the fields we turned yellow
46           for (Node node : grid.getChildren()) {
47               if (node instanceof TextField) {
48                   System.out.println(node.getStyle());
49                   // clear
50                   if (node.getStyle() != null && !node.getStyle().equals("")) {
51                       ((TextField) node).setText("");
52                       node.setStyle(null);
53                   }
54               }
55           }
56       }
57       public static String capitalize(String str) {
58           // Method to make names appear with capital first letter
59           // also turn uppercase letters not in first position into lowercase
60           // sorry McDonald!
61           if(str == null || str.isEmpty()) {
62               return str;
63           }
64           return str.substring(0, 1).toUpperCase() + str.substring(1).toLowerCase();
65       }

109      public void onAddPatient(ActionEvent actionEvent) {
110          String cprInput= cprText.getText();
111          String firstNameInput= capitalize(firstNameText.getText());
112          String lastNameInput= capitalize(lastNameText.getText());
113          boolean validInput= true;
114  
115          // validate the user input
116  
117          if (!DataValidator.isValidCPR(cprInput)){
118              // we need a CPR number in order to register a user
119              // highlight the CPR input field if content is invalid
120              cprText.setStyle("-fx-background-color: yellow");
121              validInput = false;
122  
123          }
124  
125          if (firstNameInput.isEmpty() || ! DataValidator.isValidName(firstNameInput)){
126              // We need a valid first name
127              firstNameText.setStyle("-fx-background-color: yellow");
128              validInput = false;
129          }
130  
131          if (lastNameInput.isEmpty() || ! DataValidator.isValidName(lastNameInput)){
132              // We need a valid last name
133              lastNameText.setStyle("-fx-background-color: yellow");
134              validInput = false;
135          }
136  
137          // Now register the patient in the patient register
138          // alert the user if the patient is already there
139  
140          if (validInput) {
141              Patient p = new Patient(firstNameInput,lastNameInput,cprInput);
142              try {
143                  patientRegister.addPatient(p);
144                  clearTextFields(grid);
145                  resetTextFields(grid);
146              } catch (Exception e) {
147  
148                  // The patient is already there
149                  e.printStackTrace();
150  
151                  cprText.setStyle("-fx-background-color: yellow");
152                  Alert alert = new Alert(Alert.AlertType.ERROR);
153                  //alert.setTitle("Error Dialog");
154                  alert.setHeaderText("Registration problem");
155                  alert.setContentText("CPR-number " + cprInput +" already found in register");
156  
157                  alert.showAndWait();
158                  clearTextFields(grid);
159                  resetTextFields(grid);
160              }
161  
162          } else {
163              // Something is wrong in one or more input fields
164              // Tell the user to correct input
165  
166              Alert alert = new Alert(Alert.AlertType.ERROR);
167              //alert.setTitle("Error Dialog");
168              alert.setHeaderText("Registration problem");
169              alert.setContentText("Correct highlighted fields");
170  
171  
172              alert.showAndWait();
173              resetTextFields(grid);
174          }
175          // list the patients
176          patientListTextArea.setText(patientRegister.listPatients());
177  
178      }
