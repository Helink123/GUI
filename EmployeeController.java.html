22       private void clearTextFields(GridPane grid) {
23           // Simple method to clear the text fields in a grid
24           for (Node node : grid.getChildren()) {
25               if (node instanceof TextField ) {
26                   // clear
27                   ((TextField)node).setText("");
28                   node.setStyle(null);
29               } else if (node instanceof TextArea) {
30                   ((TextArea) node).setText("");
31                   node.setStyle(null);
32               }
33           }
34       }

36       private void resetTextFields(GridPane grid) {
37           // Simple method to reset the style on text fields in a grid which have a style set
38           // and don't touch the other fields
39           // We use it to clear the fields we turned yellow
40           for (Node node : grid.getChildren()) {
41               if (node instanceof TextField) {
42                   System.out.println(node.getStyle());
43                   // clear
44                   if (node.getStyle() != null && !node.getStyle().equals("")) {
45                       ((TextField) node).setText("");
46                       node.setStyle(null);
47                   }
48               }
49           }
50       }
51       public static String capitalize(String str) {
52           // Method to make names appear with capital first letter
53           // also turn uppercase letters not in first position into lowercase
54           // sorry McDonald!
55           if(str == null || str.isEmpty()) {
56               return str;
57           }
58           return str.substring(0, 1).toUpperCase() + str.substring(1).toLowerCase();
59       }
